Alerts & Dialogues: Open from & close to a direction
 
Click to show alert 
            
       <!-- 
             Show simple Angular Material style alert message on click of the button
                          
            // Create alert message object. 
            var aSimpleAlert = $mdDialog
            .alert()
            .parent(angular.element(document.body))
            .title("Welcome ")
            .textContent(" This is a simple alert message ")
            .ok("Okay ")
            ;             
             
            // Function to demonstrate opening from and closing in a direction
            $scope.configureAndOpenAlertFromADirection = function(){
            
                // Configure alert object to open from a div element with id topRight
                aSimpleAlert.openFrom("#topRight");
                // Configure alert object to close to a div element with id topLeft        
                aSimpleAlert.closeTo("#topLeft");
            
                // Show the alert
                $mdDialog.show(aSimpleAlert);
            };          
          -->
        <md-button ng-click="configureAndOpenAlertFromADirection() " class="md-raised md-primary ">Click to show alert</md-button>
Click to show confirm dialog 
            
                <!-- 
                    On click of this button, show confirm dialog using a controller function 
                    
                    // Create confirm dialog object. 
                    var aConfirmDialog = $mdDialog
                    .confirm()
                    .parent(angular.element(document.body))
                    .title("Confirm")
                    .textContent(" Are you sure to continue?")
                    .ok("Yes")
                    .cancel("No")
                    .openFrom("#left")
                    ;          
                    
                    // Function to demonstrate opening from and closing in a direction
                    $scope.configureAndOpenDialogFromADirection = function(){
                    
                    
                        // Configure dialog object to open from a div element with id bottomLeft
                        aConfirmDialog.openFrom("#bottomLeft");
                        // Configure dialog object to close to a div element with id bottomRight        
                        aConfirmDialog.closeTo("#bottomRight");
                    
                        // Show the alert
                        $mdDialog.show(aConfirmDialog);
                    }             
                -->
                <md-button ng-click="showConfirmDialog($event)" class="md-primary">Confirm Dialog</md-button>
