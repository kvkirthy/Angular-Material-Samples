                <!-- 
                Use the directive md-tabs 
               
                md-stretch-tabs: auto
                auto - stretched to take full width on a smaller screen like mobile phone and tablet in portrait mode. It does not stretch tabs on a mobile screen in landscape mode or a desktop screen (greater screen width).
                
                md-align-tabs="top" tab titles on top.
                
                md-center-tabs: With horizontal space available, it aligns tabs to center. If we use auto or never for md-stretch-tabs, preferably set md-center-tabs.

                md-swipe-content: It allows user to swipe on content of the tab right or left to move to next or previous tabs. For a better mobile/table or any other touch screen experience, set md-swipe-content.
            -->

        <md-tabs md-stretch-tabs="auto" md-align-tabs="top" md-dynamic-height md-border-bottom md-center-tabs md-swipe-content>
